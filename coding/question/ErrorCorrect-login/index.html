Correct errors

The code below has a few errors. So that it can't work as we want.
Expected function:
- Login with username and password.
- If the username and password are correct, it will return true and output: 'Ok'.
- If the username or password is incorrect, it will return false and output: 'Wrong username or password'.

As the test result:
- We can only login in to the first account in the json.
- The username and password are correct but we can't login in to the second account.
- The output is: 'Wrong username or password'.
- The return is false.

var json=[{username:"xx", password:"xx"},{}....]; var userInput = ['Username','Password']

Code shows as below: 


function login(){
    var data = json;
    for(let i = 0; i < data.length; i++){
        if(userInput[0] == data[i].username && userInput[1] == data[i].password){
            // userInput[0] is Username, userInput[1] is Password
            console.log('Yes');
            return true;
        }
        else{
            console.log('Wrong username or password');
            return false;
        }
    }
}

login();

Tips

After return, the function will stop.

The code is here:
DO NOT ONLY COPY BUT ALSO UNDERSTAND THE CODE. 


function login(){
    var data = json;
    for(let i = 0; i < data.length; i++){
        if(json[i].username == userInput[0] && json[i].password == userInput[1]){
            // userInput[0] is Username, userInput[1] is Password
            console.log('Yes');
            return true;
        }
    }
    console.log('Wrong username or password');
    return false;
}

login();

Why? Because if we return false in the first loop, the function will stop. And it can't continue to the second loop to check the second account.
